# 5178 노드의 합



```python
def set_tree(n):
    global tr, N
    l_child = n * 2
    r_child = n * 2 + 1
    if l_child > N and r_child > N:
        return tr[n]
    elif l_child > N:
        tr[n] = set_tree(r_child)
        return tr[n]
    elif r_child > N:
        tr[n] = set_tree(l_child)
        return tr[n]
    else:
        tr[n] = set_tree(l_child) + set_tree(r_child)
        return tr[n]


for T in range(1, int(input()) + 1):
    N, M, L = map(int, input().split())
    tr = [0] * (N + 1)
    for _ in range(M):
        idx, val = map(int, input().split())
        tr[idx] = val
    set_tree(1)
    print(f'#{T} {tr[L]}')
```

```python
def set_BST(n):
    global tr, num
    if n > N:
        return
    set_BST(n * 2)
    tr[n] = num
    num += 1
    set_BST(n * 2 + 1)


for T in range(1, int(input()) + 1):
    N = int(input())
    tr = [0] * (N + 1)
    num = 1
    set_BST(1)
    print(f'#{T} {tr[1]} {tr[N//2]}')
```

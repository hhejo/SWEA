# 1979 어디에 단어가 들어갈 수 있을까



```python
for t in range(1, int(input())+1):
    n, k = map(int, input().split())
    board = [list(map(int, input().split())) for _ in range(n)]  # NxN 리스트
    r_board = list(map(list, zip(*board)))  # 전치 리스트
    word = r_word = 0  # 해당 행에 들어갈 수 있는 단어의 수
    for i in range(n):  # 행 반복
        space = r_space = 0  # 연속된 빈 칸의 수 누적 변수
        for j in range(n):  # 칸 하나씩 반복
            if board[i][j]:  # i행 j열 칸 공간 있으면 연속 빈 칸 1 증가
                space += 1
            else:  # 막혀있으면
                if space == k:  # 지금까지 k개 칸 비었으면 단어 수 1 증가
                    word += 1
                space = 0  # 막혀있으므로 초기화
            if r_board[i][j]:  # 뒤집힌 리스트에 대해서도 똑같이 시행
                r_space += 1
            else:
                if r_space == k:
                    r_word += 1
                r_space = 0
        if space == k:  # 마지막에 연속된 빈 칸의 수 k개였으면 단어 1 증가
            word += 1
        if r_space == k:  # 뒤집힌 리스트도 동일
            r_word += 1
    print(f'#{t} {word+r_word}')
```

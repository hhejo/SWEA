# 2001 파리 퇴치



## 1.

```python
for t in range(1, int(input()) + 1):
    n, m = map(int, input().split())
    board = [list(map(int, input().split())) for _ in range(n)]
    kills = max_kills = 0  # 현재 파리채 킬 수, 최대 킬 수
    for i in range(n-m+1):
        for j in range(n-m+1):
            for x in range(m):
                for y in range(m):
                    kills += board[x+i][y+j]
            max_kills = max(max_kills, kills)  # 최대 킬 수
            kills = 0  # 킬 수 초기화
    print(f'#{t} {max_kills}')
```



## 2.

```python
for t in range(1, int(input()) + 1):
    n, m = map(int, input().split())
    board = [list(map(int, input().split())) for _ in range(n)]
    kills = max_kills = 0  # 현재 파리채 킬 수, 최대 킬 수
    for i in range(n-m+1):
        for j in range(n-m+1):
            for row in range(m):
                kills += sum(board[row+i][j:j+m])  # 파리채 한 행의 킬 수 누적
            max_kills = max(max_kills, kills)  # 최대 킬 수
            kills = 0  # 킬 수 초기화
    print(f'#{t} {max_kills}')
```



## 3.

```python
for t in range(1, int(input()) + 1):
    n, m = map(int, input().split())
    board = [list(map(int, input().split())) for _ in range(n)]
    kills = []  # M x M 영역 파리채 킬 수들의 리스트
    for i in range(n-m+1):  # n-m+1번 진행
        for j in range(n-m+1):  # n-m+1번 진행
            kills.append(sum(sum(area[j:j+m]) for area in board[i:i+m]))  # 현재 파리채 킬 수 리스트에 삽입
    print(f'#{t} {max(kills)}')  # kills에서 최대 킬 수 뽑기
```

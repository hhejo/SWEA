```python
for T in range(1, 11):
    N = int(input())
    arr = list(map(int, input().split()))
    cnt = 0  # 조망권 누적 세대수
    for i in range(2, N-2):  # 2 ~ N-3
        min_cnt = arr[i]  # 최솟값 초기화
        for j in [-2, -1, 1, 2]:
            min_cnt = arr[i] - arr[i+j] if min_cnt >= arr[i] - arr[i+j] else min_cnt  # 4개 중 최솟값
        cnt += min_cnt if min_cnt > 0 else 0  # 음수이면 0
    print(f'#{T} {cnt}')
```

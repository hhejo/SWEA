# 4837 부분집합의 합



```python
from itertools import combinations

li = [*range(1, 13)]
for t in range(1, int(input()) + 1):
    n, k = map(int, input().split())
    combination_list = list(combinations(li, n))  # li에서 n개를 뽑는 조합
    cnt = 0
    for com in combination_list:  # 조합들 하나씩 반복
        if sum(com) == k:
            cnt += 1
    print(f'#{t} {cnt}')
```

# 1231 중위순회



```python
def inorder(n):
    global N, tr
    if n <= N:
        inorder(2 * n)
        print(tr[n], end='')
        inorder(2 * n + 1)


for T in range(1, 11):
    N = int(input())
    tr = [''] * (N + 1)
    for i in range(1, N + 1):
        li = list(input().split())
        tr[int(li[0])] = li[1]
    print(f'#{T} ', end='')
    inorder(1)
    print()
```

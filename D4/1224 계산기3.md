# 1224 계산기3



```python
for TC in range(1, 11):
    input()
    S = input()
    stk, exp = [], []  # 스택, 후위표기식
    # 후위표기식으로 변환
    for s in S:
        if '(' == s:
            stk.append(s)
        elif ')' == s:
            while stk and '(' != stk[-1]:
                exp.append(stk.pop())
            stk.pop()
        elif '*' == s:
            while stk and '*' == stk[-1]:
                exp.append(stk.pop())
            stk.append(s)
        elif '+' == s:
            while stk and stk[-1] in '*+':
                exp.append(stk.pop())
            stk.append(s)
        else:
            exp.append(s)
    # 후위표기식 계산
    for o in exp:
        if '*' == o:
            stk.append(str(int(stk.pop()) * int(stk.pop())))
        elif '+' == o:
            stk.append(str(int(stk.pop()) + int(stk.pop())))
        else:
            stk.append(o)
    print(f'#{TC} {stk[-1]}')
```

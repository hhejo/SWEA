# 5099 피자 굽기



## 1.

```python
for T in range(1, int(input()) + 1):
    N, M = map(int, input().split())
    pizza = list(map(int, input().split()))
    oven = [[i, pizza[i]] for i in range(N)]  # 화덕, 피자 N개로 초기화
    pizza_idx = N  # 다음에 화덕에 넣을 피자의 번호
    oven_idx = 0  # 오븐 입구 가리키는 인덱스
    cnt = N  # 현재 오븐의 피자의 개수
    ans = 0  # 정답
    flag = True
    while flag:
        # 오븐에 치즈가 다 녹은 피자가 있는지 확인
        for i in range(N):
            # 화덕에 피자가 하나만 있는지 확인
            if 1 == cnt:
                for j in range(N):
                    if oven[j]:
                        ans = oven[j][0] + 1
                        flag = False
            # 화덕 i번째 자리에 피자가 있을 때
            if oven[i]:
                # 치즈가 다 녹았을 때
                if 0 == oven[i][1]:
                    # 남은 넣을 피자가 있으면 피자 투입
                    if pizza_idx < M:
                        oven[i] = [pizza_idx, pizza[pizza_idx]]
                        pizza_idx += 1
                    # 없으면 피자 꺼냄
                    else:
                        oven[i].clear()
                        cnt -= 1
        # 피자 치즈 절반씩 녹이기
        for i in range(N):
            if oven[i]:
                oven[i][1] //= 2
    print(f'#{T} {ans}')
```



## 2.

```python
from collections import deque

for T in range(1, int(input()) + 1):
    N, M = map(int, input().split())
    pizza = list(map(int, input().split()))
    oven = deque([i, pizza[i]] for i in range(N))
    p_idx = N
    while 1 != len(oven):
        if 0 == oven[0][1]:
            if p_idx < M:
                oven.popleft()
                oven.appendleft([p_idx, pizza[p_idx]])
                p_idx += 1
            else:
                oven.popleft()
        else:
            oven[0][1] //= 2
            oven.append(oven.popleft())
    print(f'#{T} {oven[0][0] + 1}')
```
